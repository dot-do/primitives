# Examples

## Basic Usage

```typescript
import { createAISite } from 'ai-site';

const site = createAISite({
  providers: [
    { type: 'openai', apiKey: process.env.OPENAI_API_KEY }
  ],
  cache: { ttl: 3600 }
});

// Use the site instance in your application
```

## Server Component

```typescript
import { AIComponent } from 'ai-site/server';

export default function Page() {
  return (
    <div>
      <h1>AI-Powered Page</h1>
      <AIComponent 
        prompt="Generate a product description" 
        model="gpt-4"
      />
    </div>
  );
}
```
