import { compileMdx } from 'nextra/compile'
import { MDXRemote } from 'nextra/mdx-remote'
import fs from 'fs'

<README/>

export async function README() {
  const readmeContent = fs.readFileSync('packages/language-models/README.md', 'utf8')
  const rawJs = await compileMdx(`
  ${readmeContent}
  `)
  return <MDXRemote compiledSource={rawJs} components={{  }} />
}
